117 static int32_t mpu6050_reg_write(uint8_t *seq, uint8_t len)
118 {
119  int32_t ercd = E_PAR;
120 
121  emsk_imu_sensor = iic_get_dev(ACC_SENSOR_IIC_ID);
122 
123  EMSK_IMU_SENSOR_CHECK_EXP_NORTN(emsk_imu_sensor != NULL);
124 
125  /* make sure set the imu sensor's slave address */
126  emsk_imu_sensor->iic_control(IIC_CMD_MST_SET_TAR_ADDR, CONV2VOID(acc_sensor_addr));
127 
128  ercd = emsk_imu_sensor->iic_control(IIC_CMD_MST_SET_NEXT_COND, CONV2VOID(IIC_MODE_STOP));
129  ercd = emsk_imu_sensor->iic_write(seq, len);
130 
131 error_exit:
132  return ercd;
133 }

143 static int32_t mpu6050_reg_read(uint8_t seq, uint8_t *val, uint8_t len)
144 {
145  int32_t ercd = E_PAR;
146 
147  emsk_imu_sensor = iic_get_dev(ACC_SENSOR_IIC_ID);
148 
149  EMSK_IMU_SENSOR_CHECK_EXP_NORTN(emsk_imu_sensor != NULL);
150 
152  emsk_imu_sensor->iic_control(IIC_CMD_MST_SET_TAR_ADDR, CONV2VOID(acc_sensor_addr));
153  /* write register address then read register value */
154  ercd = emsk_imu_sensor->iic_control(IIC_CMD_MST_SET_NEXT_COND, CONV2VOID(IIC_MODE_RESTART));
155 
156  /* send data reading request to mpu6050 */
157  ercd = emsk_imu_sensor->iic_write(&seq, 1);
158 
159  emsk_imu_sensor->iic_control(IIC_CMD_MST_SET_NEXT_COND, CONV2VOID(IIC_MODE_STOP));
160  /* read len data from mpu6050 */
161  ercd = emsk_imu_sensor->iic_read(val, len);
162 
163 error_exit:
164  return ercd;
165 }

173 extern int32_t acc_sensor_init(uint32_t slv_addr)
174 {
175  int32_t ercd = E_OK;
176 
177  emsk_imu_sensor = iic_get_dev(ACC_SENSOR_IIC_ID);
178 
179  EMSK_IMU_SENSOR_CHECK_EXP_NORTN(emsk_imu_sensor != NULL);
180 
181  /* the device working as master */
182  ercd = emsk_imu_sensor->iic_open(DEV_MASTER_MODE, IIC_SPEED_STANDARD);
183  if ((ercd == E_OK) || (ercd == E_OPNED)) {
184  ercd = emsk_imu_sensor->iic_control(IIC_CMD_MST_SET_TAR_ADDR, CONV2VOID(slv_addr));
185  acc_sensor_addr = slv_addr;
186 
187  /* write value to mpu6050 to set registers */
188  mpu6050_reg_write(imu_init_seq1, 2);
189  mpu6050_reg_write(imu_init_seq0, 5);
190  }
191 
192 error_exit:
193  return ercd;
194 }

202 extern int32_t acc_sensor_read(acc_values* acc_temp)
203 {
204  int32_t ercd = E_OK;
205 
206  emsk_imu_sensor = iic_get_dev(ACC_SENSOR_IIC_ID);
207 
208  /* read 6 data from mpu6050 */
209  ercd = mpu6050_reg_read(MPU6050_REG_ACCEL_XOUT_H, acc_data.buf, 6);
210 
211  if (ercd != 6) {
212  ercd = E_OBJ;
213  } else {
214  acc_temp->accl_x = acc_unit * (float)(short)((acc_data.axh << 8) | acc_data.axl);
215  acc_temp->accl_y = acc_unit * (float)(short)((acc_data.ayh << 8) | acc_data.ayl);
216  acc_temp->accl_z = acc_unit * (float)(short)((acc_data.azh << 8) | acc_data.azl);
217 
218  ercd = E_OK;
219  }
220 
221 error_exit:
222  return ercd;
223 }
