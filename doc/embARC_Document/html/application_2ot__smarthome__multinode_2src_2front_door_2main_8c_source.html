231 static void lock_sta_response_send(void * p_context,
232  otCoapHeader * p_request_header,
233  const otMessageInfo * p_message_info)
234 {
235  otError error = OT_ERROR_NONE;
236  otCoapHeader header;
237  otMessage * p_response;
238 
239  do {
240  otCoapHeaderInit(&header, OT_COAP_TYPE_ACKNOWLEDGMENT, OT_COAP_CODE_CHANGED);
241  otCoapHeaderSetMessageId(&header, otCoapHeaderGetMessageId(p_request_header));
242  otCoapHeaderSetToken(&header,
243  otCoapHeaderGetToken(p_request_header),
244  otCoapHeaderGetTokenLength(p_request_header));
245 
246  p_response = otCoapNewMessage(p_context, &header);
247  if (p_response == NULL) {
248  break;
249  }
250 
251  error = otCoapSendResponse(p_context, p_response, p_message_info);
252 
253  } while (false);
254 
255  if (error != OT_ERROR_NONE && p_response != NULL) {
256  otMessageFree(p_response);
257  }
258 }

260 static void lock_sta_request_handler(void * p_context,
261  otCoapHeader * p_header,
262  otMessage * p_message,
263  const otMessageInfo * p_message_info)
264 {
265  (void)p_message;
266  char sta;
267 
268  do {
269  if (otCoapHeaderGetType(p_header) != OT_COAP_TYPE_CONFIRMABLE &&
270  otCoapHeaderGetType(p_header) != OT_COAP_TYPE_NON_CONFIRMABLE) {
271  EMBARC_PRINTF("err: lock_sta handler - bad header\r\n");
272  break;
273  }
274 
275  if (otCoapHeaderGetCode(p_header) != OT_COAP_CODE_PUT) {
276  EMBARC_PRINTF("err: lock_sta handler - not PUT request\r\n");
277  break;
278  }
279 
280  if (otMessageRead(p_message, otMessageGetOffset(p_message), &sta, 1) != 1) {
281  EMBARC_PRINTF("err: lock_sta handler - missing status\r\n");
282  }
283 
284  switch (sta) {
285  case LOCK:
286  EMBARC_PRINTF("\r\nReceived [lock status: LOCK] from UI!\r\n");
287  frontdoor_lock();
288  break;
289 
290  case UNLOCK:
291  EMBARC_PRINTF("\r\nReceived [lock status: UNLOCK] from UI!\r\n");
292  frontdoor_unlock();
293  break;
294 
295  default:
296  EMBARC_PRINTF("err: lock_sta handler - bad status\r\n");
297  break;
298  }
299 
300  if (otCoapHeaderGetType(p_header) == OT_COAP_TYPE_CONFIRMABLE) {
301  lock_sta_response_send(p_context, p_header, p_message_info);
302  }
303  } while (false);
304 }

326 static void lock_sta_request_send(otInstance * p_instance, char sta)
327 {
328  otError error = OT_ERROR_NONE;
329  otMessage * p_message;
330  otMessageInfo messageInfo;
331  otCoapHeader header;
332 
333  do {
334  otCoapHeaderInit(&header, OT_COAP_TYPE_CONFIRMABLE, OT_COAP_CODE_PUT);
335  otCoapHeaderAppendUriPathOptions(&header, "lock_sta");
336  otCoapHeaderSetPayloadMarker(&header);
337 
338  p_message = otCoapNewMessage(p_instance, &header);
339  if (p_message == NULL) {
340  EMBARC_PRINTF("err: Failed to allocate message for CoAP Request\r\n");
341  break;
342  }
343 
344  error = otMessageAppend(p_message, &sta, sizeof(sta));
345  if (error != OT_ERROR_NONE) {
346  break;
347  }
348 
349  memset(&messageInfo, 0, sizeof(messageInfo));
350  messageInfo.mInterfaceId = OT_NETIF_INTERFACE_ID_THREAD;
351  messageInfo.mPeerPort = OT_DEFAULT_COAP_PORT;
352 
353  /* Convert a human-readable IPv6 address string into a binary representation */
354  otIp6AddressFromString(GATEWAY_ADDR_USER, &messageInfo.mPeerAddr);
355 
356  error = otCoapSendRequest(p_instance,
357  p_message,
358  &messageInfo,
359  &lock_sta_response_handler,
360  p_instance);
361  EMBARC_PRINTF("Send [lock status: %c] to gateway[%s :%d]\r\n",
362  sta, GATEWAY_ADDR_USER, OT_DEFAULT_COAP_PORT);
363  } while (false);
364 
365  if (error != OT_ERROR_NONE && p_message != NULL) {
366  EMBARC_PRINTF("err: Failed to send CoAP Request: %d\r\n", error);
367  otMessageFree(p_message);
368  }
369 }
