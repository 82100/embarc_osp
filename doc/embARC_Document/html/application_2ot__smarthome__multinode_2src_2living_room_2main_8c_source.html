296 static void light_sta_response_send(void * p_context,
297  otCoapHeader * p_request_header,
298  const otMessageInfo * p_message_info)
299 {
300  otError error = OT_ERROR_NONE;
301  otCoapHeader header;
302  otMessage * p_response;
303 
304  do {
305  otCoapHeaderInit(&header, OT_COAP_TYPE_ACKNOWLEDGMENT, OT_COAP_CODE_CHANGED);
306  otCoapHeaderSetMessageId(&header, otCoapHeaderGetMessageId(p_request_header));
307  otCoapHeaderSetToken(&header,
308  otCoapHeaderGetToken(p_request_header),
309  otCoapHeaderGetTokenLength(p_request_header));
310 
311  p_response = otCoapNewMessage(p_context, &header);
312  if (p_response == NULL) {
313  break;
314  }
315 
316  error = otCoapSendResponse(p_context, p_response, p_message_info);
317 
318  } while (false);
319 
320  if (error != OT_ERROR_NONE && p_response != NULL) {
321  otMessageFree(p_response);
322  }

325 static void light_sta_request_handler(void * p_context,
326  otCoapHeader * p_header,
327  otMessage * p_message,
328  const otMessageInfo * p_message_info)
329 {
330  (void)p_message;
331  char sta;
332 
333  do {
334  if (otCoapHeaderGetType(p_header) != OT_COAP_TYPE_CONFIRMABLE &&
335  otCoapHeaderGetType(p_header) != OT_COAP_TYPE_NON_CONFIRMABLE) {
336  EMBARC_PRINTF("err: light_sta handler - bad header\r\n");
337  break;
338  }
339 
340  if (otCoapHeaderGetCode(p_header) != OT_COAP_CODE_PUT) {
341  EMBARC_PRINTF("err: light_sta handler - not PUT request\r\n");
342  break;
343  }
344 
345  if (otMessageRead(p_message, otMessageGetOffset(p_message), &sta, 1) != 1) {
346  EMBARC_PRINTF("err: light_sta handler - missing status\r\n");
347  }
348 
349  switch (sta) {
350  case LIGHT_ON:
351  EMBARC_PRINTF("\r\nReceived [light status: ON] from UI!\r\n");
352  livingroom_light_on();
353  break;
354 
355  case LIGHT_OFF:
356  EMBARC_PRINTF("\r\nReceived [light status: OFF] from UI!\r\n");
357  livingroom_light_off();
358  break;
359 
360  default:
361  EMBARC_PRINTF("err: light_sta handler - bad status\r\n");
362  break;
363  }
364 
365  if (otCoapHeaderGetType(p_header) == OT_COAP_TYPE_CONFIRMABLE) {
366  light_sta_response_send(p_context, p_header, p_message_info);
367  }
368  } while (false);
369 }

391 static void light_sta_request_send(otInstance * p_instance, char sta)
392 {
393  otError error = OT_ERROR_NONE;
394  otMessage * p_message;
395  otMessageInfo messageInfo;
396  otCoapHeader header;
397 
398  do {
399  otCoapHeaderInit(&header, OT_COAP_TYPE_CONFIRMABLE, OT_COAP_CODE_PUT);
400  otCoapHeaderAppendUriPathOptions(&header, "light_sta");
401  otCoapHeaderSetPayloadMarker(&header);
402 
403  p_message = otCoapNewMessage(p_instance, &header);
404  if (p_message == NULL) {
405  EMBARC_PRINTF("err: Failed to allocate message for CoAP Request\r\n");
406  break;
407  }
408 
409  error = otMessageAppend(p_message, &sta, sizeof(sta));
410  if (error != OT_ERROR_NONE) {
411  break;
412  }
413 
414  memset(&messageInfo, 0, sizeof(messageInfo));
415  messageInfo.mInterfaceId = OT_NETIF_INTERFACE_ID_THREAD;
416  messageInfo.mPeerPort = OT_DEFAULT_COAP_PORT;
417 
418  /* Convert a human-readable IPv6 address string into a binary representation */
419  otIp6AddressFromString(GATEWAY_ADDR_USER, &messageInfo.mPeerAddr);
420 
421  error = otCoapSendRequest(p_instance,
422  p_message,
423  &messageInfo,
424  &light_sta_response_handler,
425  p_instance);
426  EMBARC_PRINTF("Send [light status: %c] to gateway[%s :%d]\r\n",
427  sta, GATEWAY_ADDR_USER, OT_DEFAULT_COAP_PORT);
428  } while (false);
429 
430  if (error != OT_ERROR_NONE && p_message != NULL) {
431  EMBARC_PRINTF("err: Failed to send CoAP Request: %d\r\n", error);
432  otMessageFree(p_message);
433  }
434 }

499 static void request_send_scan(void)
500 {
501  int32_t val = 0;
502  char *temp = "0";
503 
504  if (flag_send_light_sta == REQUEST_SEND) {
505  flag_send_light_sta = REQUEST_NONE;
506  light_sta_request_send(m_app.p_ot_instance, light_sta);
507  }
508 
509  if (flag_send_temp == REQUEST_SEND) {
510  flag_send_temp = REQUEST_NONE;
511 
512  if (E_OK != temp_sensor_read(&val)) {
513  EMBARC_PRINTF("err: Failed to read temperature\r\n");
514  return;
515  }
516  itoa(val, temp, 10);
517  EMBARC_PRINTF("Temperature in the LivingRoom now: %d.%d`C\r\n", val / 10, val % 10);
518  temp_request_send(m_app.p_ot_instance, temp);
519  led_temp_blink();
520  }
521 }
