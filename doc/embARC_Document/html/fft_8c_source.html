68 {

70  compx t, ws;
71  for (L = 1; L <= M; L++) {
72  le = (2 << L) >> 1;
73  B = le / 2;
74 
75  for (j = 0; j <= B - 1; j++) {
76  p = (((2 << M) >> L) >> 1) * j;
77  ws.real = cos_tab[p];
78  ws.imag = sin_tab[p];
79 
80  for (i = j; i <= fft_N - 1; i = i + le) {
81  ip = i + B;
82  t.real = xin[ip].real * ws.real - xin[ip].imag * ws.imag;
83  t.imag = xin[ip].real * ws.imag + xin[ip].imag * ws.real;
84  xin[ip].real = xin[i].real - t.real;
85  xin[ip].imag = xin[i].imag - t.imag;
86  xin[i].real = xin[i].real + t.real;
87  xin[i].imag = xin[i].imag + t.imag;
88  }
