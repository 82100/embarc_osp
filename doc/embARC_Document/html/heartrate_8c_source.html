198 static int32_t max30102_reg_write(uint8_t* seq, uint8_t len)
199 {
200  int32_t ercd = E_PAR;
201 
202  emsk_max_sensor = iic_get_dev(HRATE_SENSOR_IIC_ID);
203 
204  EMSK_HEART_RATE_SENSOR_CHECK_EXP_NORTN(emsk_max_sensor != NULL);
205 
206  /* make sure set the max sensor's slave address */
207  emsk_max_sensor->iic_control(IIC_CMD_MST_SET_TAR_ADDR, CONV2VOID(hrate_sensor_addr));
208 
209  ercd = emsk_max_sensor->iic_control(IIC_CMD_MST_SET_NEXT_COND, CONV2VOID(IIC_MODE_STOP));
210  ercd = emsk_max_sensor->iic_write(seq, len);
211 
212 error_exit:
213  return ercd;
214 }

224 static int32_t max30102_reg_read(uint8_t seq, uint8_t* val, uint8_t len)
225 {
226  int32_t ercd = E_PAR;
227 
228  emsk_max_sensor = iic_get_dev(HRATE_SENSOR_IIC_ID);
229 
230  EMSK_HEART_RATE_SENSOR_CHECK_EXP_NORTN(emsk_max_sensor != NULL);

233  emsk_max_sensor->iic_control(IIC_CMD_MST_SET_TAR_ADDR, CONV2VOID(hrate_sensor_addr));
234  /* write register address then read register value */
235  ercd = emsk_max_sensor->iic_control(IIC_CMD_MST_SET_NEXT_COND, CONV2VOID(IIC_MODE_RESTART));

243 error_exit:

253 extern int32_t hrate_sensor_init(uint32_t slv_addr)
254 {
255  int32_t ercd = E_OK;
256 
257  emsk_max_sensor = iic_get_dev(HRATE_SENSOR_IIC_ID);
258 
259  EMSK_HEART_RATE_SENSOR_CHECK_EXP_NORTN(emsk_max_sensor != NULL);
260 
261  if ((ercd == E_OK) || (ercd == E_OPNED)) {
262  ercd = emsk_max_sensor->iic_control(IIC_CMD_MST_SET_TAR_ADDR, CONV2VOID(slv_addr));
263  hrate_sensor_addr = slv_addr;
264 
265  /* write value to max30102 to set registers */
266  max30102_reg_write(hrate_int_enable, 3);
267  max30102_reg_write(hrate_init_fifo_seq, 4);
268  max30102_reg_write(hrate_init_seq, 6);
269  }
270 
271 error_exit:
272  return ercd;
273 }

281 extern int32_t hrate_sensor_read(int32_t* hrate)
282 {
283  int32_t ercd = E_OK;
284 
285  /* int_rdy the interrupt flag to determine whether to read data */
286  uint8_t int_rdy = 0;
287 
288  emsk_max_sensor = iic_get_dev(HRATE_SENSOR_IIC_ID);
289 
290  EMSK_HEART_RATE_SENSOR_CHECK_EXP_NORTN(emsk_max_sensor != NULL);
291 
292  /* get the interrupt flag */
293  max30102_reg_read(MAX30102_REG_INT_STATUS_1, &int_rdy, 1);
294 
295  /* MAX30102_STATUS_PPG_RDY: New FIFO Data Ready */
296  if (int_rdy & MAX30102_STATUS_PPG_RDY)
297  {
298  /* read 3 data from max30102 */
299  ercd = max30102_reg_read(MAX30102_REG_FIFO_DATA, hrate_data.buf, 3);
300 
301  if (ercd != 3) {
302  ercd = E_OBJ;
303  } else {
304  ercd = E_OK;
305  *hrate = ((hrate_data.red_h && 0x3) << 16) |
306  (hrate_data.red_m << 8) | hrate_data.red_l;
307  }
308  int_rdy = 0;
309  }
310 
311 error_exit:
312  return ercd;
313 }
