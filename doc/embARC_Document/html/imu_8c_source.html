204 void imu_acc_feature_get(int32_t *point, float multi_throd, uint32_t *recognition, uint8_t *peak_order, uint8_t peak_flag)
205 {
206  uint8_t i;
207  int32_t max_value = 0, min_value = 0, last_value_flag = 0;
208  int32_t *cal_point = point, posi_zero_value, nega_zero_value;
209 
210  *recognition = 0;
211  /* Get the the maximum value and minimum value */
212  for (i = ACC_DATA_CON_NUM - 1; i > ACC_BUF_WINDOW_NUM; i--) {
213  if (*cal_point > max_value) {
214  max_value = *cal_point;
215  }
216  if (*cal_point < min_value) {
217  min_value = *cal_point;
218  }
219  cal_point--;
220  }
221 
222  /* Point the pointer to the first address of the acceleration cache */
223  cal_point += (ACC_DATA_CON_NUM - 1 - ACC_BUF_WINDOW_NUM);
224  /* Get the first positive peak's position */
225  if (peak_flag == GET_PEAK_PO) {
226  for (i = ACC_DATA_CON_NUM - 1; i > ACC_BUF_WINDOW_NUM; i--) {
227  /* If the value is bigger than 0.8 times of the maximum,
228  calibrate the first positive peak's position */
229  if (*cal_point >= (max_value * 0.8)) {
230  *peak_order = i;
231  break;
232  }
233  cal_point--;
234  }
235  } else {
236  for (i = ACC_DATA_CON_NUM - 1; i > ACC_BUF_WINDOW_NUM; i--) {
237  if ((*cal_point) <= (min_value * 0.8)) {
238  /* If the value is smaller than 0.8 times of the minimal,
239  calibrate the first negative peak's position */
240  *peak_order = i;
241  break;
242  }
243  cal_point--;
244  }
245  }
246 
247  /* Set the threshold for determining acceleration 0 */
248  posi_zero_value = 0.1 * max_value;
249  nega_zero_value = 0.1 * min_value;
250  /* Set the positive and negative thresholds.If the max_value*multi_throd is smaller
251  than JUDGE_SYMBOL_THRESHOLD,the threshold will be set as JUDGE_SYMBOL_THRESHOLD */
252  max_value = (max_value * multi_throd > JUDGE_SYMBOL_THRESHOLD) ? max_value * multi_throd : JUDGE_SYMBOL_THRESHOLD;
253  min_value = (min_value * multi_throd < -JUDGE_SYMBOL_THRESHOLD) ? min_value * multi_throd : -JUDGE_SYMBOL_THRESHOLD;
254 
255  /* Judge the positive and negative of the intercept waveform
256  If the judgment waveform is positive，the variable "(*recognition)" is shifted by two bits and
257  takes 0x02.Take "last_value_flag" with 0x02 \
258  If the judgment waveform is negative，the variable "(*recognition)" is shifted by two bits and
259  takes 0x01.Take "last_value_flag" with 0x01 \
260  Take "last_value_flag" with 0x00
261  */
262  for (i = ACC_DATA_CON_NUM - 1; i >= ACC_BUF_WINDOW_NUM; i--)
263  {
264  /* If the last feature is not negative(Judge by the second bit of last_value_flag,if the second
265  bit of last_value_flag is not one),\
266  and the value is smaller than min_value */
267  if (((*point) < min_value) && ((last_value_flag & 0x02) == 0x00)) {
268  (*recognition) = ((*recognition) << 2) | PEAK_WAVE_NEGATIVE;
269  last_value_flag &= 0x00;
270  last_value_flag |= PEAK_WAVE_NEGATIVE;
271  }
272 
273  /* If the last feature is not zero(Judge by the first and second bits of last_value_flag,if these
274  bits of last_value_flag are zero) */
275  if ((last_value_flag & 0x03) != 0x00) {
276  if (((*point) > nega_zero_value) && (*point) < posi_zero_value) {
277  last_value_flag &= 0x00;
278  }
279  }
280 
281  /* If the last feature is not positive(Judge by the first bit of last_value_flag,if the first bit
282  of last_value_flag is not one) */
283  if (((*point) > max_value) && (((*recognition) & 0x01) == 0x00)) {
284  (*recognition) = ((*recognition) << 2) | PEAK_WAVE_POSITIVE;
285  last_value_flag &= 0x00;
286  last_value_flag |= PEAK_WAVE_POSITIVE;
287  }
288 
289  point--;
290  }
291 }

297 void imu_mpu6050_update(void)
298 {
299  int8_t i;
300  static uint8_t filter_cnt = 0; /* Filter cache flag */
301  //float scope_data[4]; /* Ued in scope */
302  uint32_t cal_acc_temp[3], acc_sum;
303  uint32_t acc_dif_temp = 0;
304  int acc_temp[3] = {0};
305 
306  acc_upbuf.acc_x_buf[filter_cnt] = mpu6050_get_data(ACCEL_XOUT_H);
307  acc_upbuf.acc_y_buf[filter_cnt] = mpu6050_get_data(ACCEL_YOUT_H);
308  acc_upbuf.acc_z_buf[filter_cnt] = mpu6050_get_data(ACCEL_ZOUT_H);
309 
310  for (i = 0; i < ACC_FILTER_NUM; i++) {
311  acc_temp[0] += acc_upbuf.acc_x_buf[i];
312  acc_temp[1] += acc_upbuf.acc_y_buf[i];
313  acc_temp[2] += acc_upbuf.acc_z_buf[i];
314  }
315 
316  acc_temp[0] = acc_temp[0] / ACC_FILTER_NUM;
317  acc_temp[1] = acc_temp[1] / ACC_FILTER_NUM;
318  acc_temp[2] = acc_temp[2] / ACC_FILTER_NUM;
319 
320  imu_mpu6050_update_ptr->acc_x = 16 * ((acc_temp[0]) * 1000) / (32768);
321  imu_mpu6050_update_ptr->acc_y = 16 * ((acc_temp[1]) * 1000) / (32768);
322  imu_mpu6050_update_ptr->acc_z = 16 * ((acc_temp[2]) * 1000) / (32768);
323 
324  /* Upte those buffer which is update dynamically */
325  for (i = ACC_BUF_WINDOW_NUM - 1; i > 0; i--) {
326  imu_mpu6050_update_ptr->acc_judge_x_buf[i] = imu_mpu6050_update_ptr->acc_judge_x_buf[i - 1];
327  }
328  imu_mpu6050_update_ptr->acc_judge_x_buf[0] = imu_mpu6050_update_ptr->acc_x;
329  for (i = ACC_BUF_WINDOW_NUM - 1; i > 0; i--) {
330  imu_mpu6050_update_ptr->acc_judge_y_buf[i] = imu_mpu6050_update_ptr->acc_judge_y_buf[i - 1];
331  }
332  imu_mpu6050_update_ptr->acc_judge_y_buf[0] = imu_mpu6050_update_ptr->acc_y;
333  for (i = ACC_BUF_WINDOW_NUM - 1; i > 0; i--) {
334  imu_mpu6050_update_ptr->acc_judge_z_buf[i] = imu_mpu6050_update_ptr->acc_judge_z_buf[i - 1];
335  }
336  imu_mpu6050_update_ptr->acc_judge_z_buf[0] = imu_mpu6050_update_ptr->acc_z;
337 
338  /* If the statr of action is captured ,store date in buffer */
339  if (partern_rec_startflag & 0x01) {
340  for (i = ACC_DATA_CON_NUM - 1; i > ACC_BUF_WINDOW_NUM; i--) {
341  imu_mpu6050_update_ptr->acc_judge_y_buf[i] = imu_mpu6050_update_ptr->acc_judge_y_buf[i - 1];
342  imu_mpu6050_update_ptr->acc_judge_x_buf[i] = imu_mpu6050_update_ptr->acc_judge_x_buf[i - 1];
343  imu_mpu6050_update_ptr->acc_judge_z_buf[i] = imu_mpu6050_update_ptr->acc_judge_z_buf[i - 1];
344  }
345  imu_mpu6050_update_ptr->acc_judge_y_buf[ACC_BUF_WINDOW_NUM] = imu_mpu6050_update_ptr->acc_judge_y_buf[ACC_BUF_WINDOW_NUM - 1] * 4;
346  imu_mpu6050_update_ptr->acc_judge_x_buf[ACC_BUF_WINDOW_NUM] = imu_mpu6050_update_ptr->acc_judge_x_buf[ACC_BUF_WINDOW_NUM - 1];
347  imu_mpu6050_update_ptr->acc_judge_z_buf[ACC_BUF_WINDOW_NUM] = imu_mpu6050_update_ptr->acc_judge_z_buf[ACC_BUF_WINDOW_NUM - 1] + 1000;
348  }
349 
350  cal_acc_temp[0] = (imu_mpu6050_update_ptr->acc_judge_x_buf[1] - imu_mpu6050_update_ptr->acc_judge_x_buf[0]);
351  cal_acc_temp[1] = (imu_mpu6050_update_ptr->acc_judge_y_buf[1] - imu_mpu6050_update_ptr->acc_judge_y_buf[0]);
352  cal_acc_temp[2] = (imu_mpu6050_update_ptr->acc_judge_z_buf[1] - imu_mpu6050_update_ptr->acc_judge_z_buf[0]);
353 
354  acc_upbuf.acc_dif_buf[filter_cnt] = abs(cal_acc_temp[0]) + abs(cal_acc_temp[1]) + abs(cal_acc_temp[2]);
355 
356  for (i = 0; i < ACC_FILTER_NUM; i++) {
357  acc_dif_temp += acc_upbuf.acc_dif_buf[i];
358  }
359  acc_dif_temp = acc_dif_temp / ACC_FILTER_NUM;
360  for (i = ACC_DATA_CON_NUM - 1; i > 0; i--) {
361  imu_mpu6050_update_ptr->acc_judge_dif_buf[i] = imu_mpu6050_update_ptr->acc_judge_dif_buf[i - 1];
362  }
363  imu_mpu6050_update_ptr->acc_judge_dif_buf[0] = acc_dif_temp;
364 
365  filter_cnt++;
366 
367  if (filter_cnt == ACC_FILTER_NUM) {
368  filter_cnt = 0;
369  }
370 }

408 uint8_t imu_pattern_recongnition(void)
409 {
410  uint8_t i, y_order, z_order;
411  static uint8_t start_action_get_flag;
412  uint32_t recognition_z = 0, recognition_y = 0;
413  uint8_t action_result = 0;
414  imu_get_stastp_point();
415 
416  if (partern_rec_startflag & 0x80) {
417  imu_acc_feature_get(&(imu_mpu6050_update_ptr->acc_judge_y_buf[ACC_DATA_CON_NUM - 1]), JUDGE_WAVE_NE_THRESHOLD, \
418  &recognition_y, &y_order, GET_PEAK_PO);
419  imu_acc_feature_get(&(imu_mpu6050_update_ptr->acc_judge_z_buf[ACC_DATA_CON_NUM - 1]), JUDGE_WAVE_PO_THRESHOLD, \
420  &recognition_z, &z_order, GET_PEAK_NE);
421 
422  /* Only when the system does not enter the mode of switching state,we will judge whether we have got the T */
423  if (start_action_get_flag == 0) {
424  /* Judge T! Z:6,2,38. Y:9,38 */
425  if ((recognition_z == WAVE_PN) || (recognition_z == WAVE_N) || (recognition_z == WAVE_NPN)) {
426  if ((recognition_y == WAVE_NP) || (recognition_y == WAVE_NPN)) {
427  /* Judge first action is vertical or vertical to reduce miscarriage of justice */
428  if (y_order > z_order) {
429  start_action_get_flag = 0x01;
430  EMBARC_PRINTF("Recognition OK!! :T!\r\n");
431  action_result = ACTION_START;
432  }
433  }
434  }
435  }
436 
437  /* Only when the system enter the mode of switching state,we will judge left or right */
438  if (start_action_get_flag != 0) {
439  if ((recognition_y == WAVE_NP) || (recognition_y == WAVE_NPN)) {
440  if ((recognition_z == 0) || (recognition_z == WAVE_P) || (recognition_z == WAVE_N)) {
441  action_result = ACTION_LEFT;
442  EMBARC_PRINTF("Recognition OK!! :LEFT\r\n");
443  }
444  }
445  if ((recognition_y == WAVE_PN) || (recognition_y == WAVE_PNP)) {
446  if ((recognition_z == 0) || (recognition_z == WAVE_P) || (recognition_z == WAVE_N)) {
447  action_result = ACTION_RIGHT;
448  EMBARC_PRINTF("Recognition OK!! :RIGHT\r\n");
449  }
450  }
451  }
452 
453  if (start_action_get_flag != 0) {
454  if ((recognition_z == WAVE_PNP) || (recognition_z == WAVE_NP)) {
455  if ((recognition_y == WAVE_NP)) {
456  action_result = ACTION_OVER;
457  start_action_get_flag = 0;
458  }
459  }
460  }
461 
462  partern_rec_startflag &= 0x7f;
463  for (i = 0; i < ACC_DATA_CON_NUM; i++)
464  {
465  imu_mpu6050_update_ptr->acc_judge_x_buf[i] = 0;
466  imu_mpu6050_update_ptr->acc_judge_y_buf[i] = 0;
467  imu_mpu6050_update_ptr->acc_judge_z_buf[i] = 0;
468 
469  }
470  }
471  return action_result;
472 }
