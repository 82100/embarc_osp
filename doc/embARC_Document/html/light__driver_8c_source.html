252 void light_ctr_rgb(uint64_t mask, uint16_t row, uint8_t red, uint8_t green, uint8_t blue)
253 {
254  uint8_t i;
255  uint32_t colordata = 0;
256  color_row = row;
257  RGB_T *ilight_color;
258 
259  if (row == LIGHT_ROW_MIDDLE) {
260  light_num = LIGHT_NUM_MIDDLE;
261  ilight_color = &ilight_middle[0];
262  piLight = ilight_middle;
263  } else {
264  light_num = LIGHT_NUM_SIDE;
265  ilight_color = &ilight_side[0];
266  piLight = ilight_side;
267  }
268 
269  /* Place the data of RGB colors into a 32-bit variable */
270  colordata |= green;
271  colordata = (colordata << 8) | red;
272  colordata = (colordata << 8) | blue;
273 
274 
275  /* Update the array element data */
276  /* In order to avoid the shift operation is optimized,set the 64 bit one */
277  for (i = 0; i < light_num; i++) {
278  if ((mask >> i) & 0x01) {
279  ilight_color[i].color = colordata;
280  }
281  }
282  xQueueSendToFront(led_queue, (void *) & piLight, (TickType_t)0);
283  delay_10us(5);
284 }
