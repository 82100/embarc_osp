116 static uint8_t mpu6050_read_reg(uint8_t regaddr)

134 void mpu6050_init(uint8_t gyro_rng, uint8_t accel_rng)
135 {
136  iic_mpu6050_int();
137  /* Reset MPU6050 */
138  mpu6050_write_reg(MPU_PWR_MGMT1_REG, 0X80);
139  vTaskDelay(100);
140  /* Wake up MPU6050 */
141  mpu6050_write_reg(MPU_PWR_MGMT1_REG, 0x07);
142  /* Set sampling frequency divider */
143  mpu6050_write_reg(SMPLRT_DIV, 0x06);
144  /* Set the low-pass filter */
145  mpu6050_write_reg(LPF_CONFIG, 0x06);
146  /* Turn off all interrupts */
147  mpu6050_write_reg(MPU_INT_EN_REG, 0x00);
148  /* Turn off IIC master mode */
149  mpu6050_write_reg(MPU_USER_CTRL_REG, 0X00);
150  /* Close fifo */
151  mpu6050_write_reg(MPU_FIFO_EN_REG, 0X00);
152  /* Set interrupt pin active low */
153  mpu6050_write_reg(MPU_INTBP_CFG_REG, 0X80);
154  /* Set CLKSEL, PLL, X axis for reference */
155  mpu6050_write_reg(MPU_PWR_MGMT1_REG, 0X01);
156  /* Enable accelerometer and gyroscope */
157  mpu6050_write_reg(MPU_PWR_MGMT2_REG, 0X00);
158  /* Configure the gyroscope full scale */
159  mpu6050_write_reg(MPU_GYRO_CFG_REG, (gyro_rng << 3));
160  /* Configure the accelerometer full scale */
161  mpu6050_write_reg(MPU_ACCEL_CFG_REG, (accel_rng << 3));
162 }

180 int mpu6050_get_data(uint8_t regaddr)
181 {
182  int data;
183  uint16_t data_temp;
184  uint8_t H = 0, L = 0;
185 
186  /* Read high 8 bits data */
187  H = mpu6050_read_reg(regaddr);
188 
189  /* Read low 8 bits data */
190  L = mpu6050_read_reg(regaddr + 1);
191  data_temp = (H << 8) | L;
192 
193  /* Change the data into a signed integer */
194  if (data_temp < 32768) {
195  data = data_temp;
196  } else {
197  data = -(65535 - data_temp);
198  }
199  return data;
200 }
