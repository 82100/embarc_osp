296  mrf24j40_write_long_ctrl_reg(MRF24J40_RFCON0, MRF24J40_CHANNEL(0) | MRF24J40_RFOPT(0x03));
297  mrf24j40_write_long_ctrl_reg(MRF24J40_RFCON1, MRF24J40_VCOOPT(0x02)); // Initialize VCOOPT = 0x02

302  mrf24j40_write_long_ctrl_reg(MRF24J40_SLPCON1, MRF24J40_SLPCLKDIV(1) | MRF24J40_CLKOUTDIS);
303 
304  mrf24j40_write_short_ctrl_reg(MRF24J40_RXFLUSH, (MRF24J40_WAKEPAD | MRF24J40_WAKEPOL));
305 
306  mrf24j40_write_short_ctrl_reg(MRF24J40_RXMCR, mrf24j40_read_short_ctrl_reg(MRF24J40_RXMCR) & (~MRF24J40_NOACKRSP));
307  // mrf24j40_write_short_ctrl_reg(MRF24J40_RXMCR, mrf24j40_read_short_ctrl_reg(MRF24J40_RXMCR) | MRF24J40_PANCOORD);

312  mrf24j40_write_short_ctrl_reg(MRF24J40_ORDER, 0xFF);
313 
314  mrf24j40_write_short_ctrl_reg(MRF24J40_BBREG1, 0x0);
315  mrf24j40_write_short_ctrl_reg(MRF24J40_BBREG2, MRF24J40_CCAMODE(0x02) | MRF24J40_CCASTH(0x00));
316  mrf24j40_write_short_ctrl_reg(MRF24J40_CCAEDTH, 0x60);

319  // TURNTIME Defaule value: 0x4 and TURNTIME=0x3

321 
322  //mrf24j40_write_short_ctrl_reg(MRF24J40_TXPEND, mrf24j40_read_short_ctrl_reg(MRF24J40_TXPEND) & MRF24J40_FPACK);
323 
324  //mrf24j40_write_long_ctrl_reg(MRF24J40_SLPCON0, MRF24J40_SLPCLKEN);
325 
326  // mrf24j40_write_short_ctrl_reg(MRF24J40_ACKTMOUT, mrf24j40_read_short_ctrl_reg(MRF24J40_ACKTMOUT) | MRF24J40_DRPACK);
327 
328  // mrf24j40_write_short_ctrl_reg(MRF24J40_RXMCR, mrf24j40_read_short_ctrl_reg(MRF24J40_RXMCR) | MRF24J40_PROMI);
329 
330  mrf24j40_ie();
331  mrf24j40_write_long_ctrl_reg(MRF24J40_RFCON3, 0x0);
332

371 /* Modified for embARC */
372 void mrf24j40_txpkt(uint8_t *frame, int16_t hdr_len, int16_t sec_hdr_len, int16_t payload_len) {
373  int16_t frame_len = hdr_len + sec_hdr_len + payload_len;
374 
375  uint8_t w = mrf24j40_read_short_ctrl_reg(MRF24J40_TXNCON);
376  w &= ~(MRF24J40_TXNSECEN);
377 
378  if (IEEE_802_15_4_HAS_SEC(frame[0])) {
379  w |= MRF24J40_TXNSECEN;
380  }
381 
382  if (IEEE_802_15_4_WANTS_ACK(frame[0])) {
383  w |= MRF24J40_TXNACKREQ;
384  }
385 
386  pmrf_txpkt_frame_write(frame, hdr_len, frame_len);
387 
388  mrf24j40_write_short_ctrl_reg(MRF24J40_TXNCON, w | MRF24J40_TXNTRIG);
389 }

459 /* Modified for embARC */
460 int16_t mrf24j40_rxpkt_intcb(uint8_t *buf, uint8_t *plqi, uint8_t *prssi) {
461  mrf24j40_write_short_ctrl_reg(MRF24J40_BBREG1, mrf24j40_read_short_ctrl_reg(MRF24J40_BBREG1) | MRF24J40_RXDECINV);
462 
463  /* Packet Reception
464  * Frame Length -- Header -- Data Payload -- FCS -- LQI -- RSSI
465  */
466  uint8_t flen = mrf24j40_read_long_ctrl_reg(MRF24J40_RXFIFO);
467 
468  if (flen > (MRF24J40_RXFIFO_SIZE - 3))
469  {
470  flen = MRF24J40_RXFIFO_SIZE - 3;
471  }
472 
473  pmrf_rxpkt_intcb_frame_read(buf, flen);
474 
475  uint8_t lqi = mrf24j40_read_long_ctrl_reg(MRF24J40_RXFIFO + flen + 1);
476  uint8_t rssi = mrf24j40_read_long_ctrl_reg(MRF24J40_RXFIFO + flen + 2);
477 
478  if (plqi != NULL) {
479  *plqi = lqi;
480  }
481 
482  if (prssi != NULL) {
483  *prssi = rssi;
484  }
485 
486  mrf24j40_rxfifo_flush();
487  mrf24j40_write_short_ctrl_reg(MRF24J40_BBREG1, mrf24j40_read_short_ctrl_reg(MRF24J40_BBREG1) & ~MRF24J40_RXDECINV);
488 
489  return (int16_t)flen;
490 }

498 int16_t mrf24j40_int_tasks(void) {

506

513  }
