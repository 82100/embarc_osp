136 static void task_sleep_monitor(void *par)
137 {
138  uint32_t svm_val; /* SVM : signal vector magnitude for difference */
139  uint32_t cnt_aw = 0; /* executing algorithm counter */
140  uint32_t cnt_sl = 0; /* counter for sleep monitoring */
141  uint32_t cnt_btemp = 0; /* counter for body temperature sampling */
142  acc_values acc_vals; /* accleration storage */
143 
144  while (1) {
145  /* read acceleration data */
146  acc_sensor_read(&acc_vals);
147 
148  /* process acceleration data and get SVM(representation of motion intensity) in 1 sample time */
149  svm_val = process_acc(acc_vals);
150 
151  /* awake event detecting algorithm */
152  if (cnt_aw < THOLD_CNT_AW) {
153  sum_svm_5s += svm_val; /* summation of SVM in 5s */
154  cnt_aw++;
155  } else {
156  /* detect awake event */
157  data_report_wn.event_awake = func_detect_awake(sum_svm_5s);
158 
159  /* print out messages for primary function */
160 #if PRINT_DEBUG_FUNC
161  print_msg_func();
162 #endif/* PRINT_DEBUG_FUNC */
163 
164  sum_svm_5s = 0;
165  cnt_aw = 0;
166  }
167 
168  /* sleep monitoring algorithm based on indigital integration method */
169  if (cnt_sl < THOLD_CNT_SL) {
170  data_report_wn.motion_intensity += svm_val; /* summation of SVM in 1 min */
171  cnt_sl++;
172  } else {
173  /* sleep-wake state monitoring */
174  data_report_wn.state = func_detect_state(data_report_wn.motion_intensity);
175 
176  data_report_wn.motion_intensity = 0;
177  cnt_sl = 0;
178  }
179 
180  data_report_wn.warn_downward = false;
181 
182  /* detect sleep downward event */
183  data_report_wn.warn_downward = func_detect_downward(acc_vals.accl_z);
184 
185 
186  /* read body temperature data */
187  if (cnt_btemp < THOLD_CNT_BTEMP) {
188  cnt_btemp++;
189  } else {
190  btemp_sensor_read(&data_report_wn.btemp);
191 
192  data_report_wn.warn_btemp = false;
193  if (data_report_wn.btemp > WARN_BTEMP_H || data_report_wn.btemp < WARN_BTEMP_L) {
194  data_report_wn.warn_btemp = true;
195  }
196 
197  cnt_btemp = 0;
198  }
199 
200  vTaskDelay(TIME_DELAY_ACC_MS);
201  }
202 }

205 static int filter_acc(int val_new,
206  int val_old,
207  bool *flag_old,
208  char *cnt,
209  unsigned char *par)
210 {
211  int val_diff;
212  bool flag_new;
213 
214  /* calculate low pass filter coefficient for x axis acceleration */
215  val_diff = val_new - val_old;
216 
217  if (val_diff > 0) {
218  flag_new = true;
219  } else {
220  flag_new = false;
221  }
222 
223  if (flag_new == *flag_old) {
224  (*cnt)++;
225  if (fabs(val_diff) > THOLD_ACC_DIFF) {
226  *cnt += CNT_ACC_STEP;
227  }
228  if (*cnt > THOLD_ACC_CNT) {
229  *par += PAR_ACC_STEP;
230  *cnt = 0;
231  }
232  } else {
233  *cnt = 0;
234  *par = PAR_ACC_BASE;
235  }
236 
237  *flag_old = flag_new;
238 
239  /* calculate output of low pass filter */
240  if (!val_diff) {
241  val_new = val_old;
242  } else {
243  val_new = val_old + val_diff * (*par) / 256;
244  }
245 
246  return val_new;
247 }

250 static int filter_svm(int val_new,
251  int val_old,
252  bool *flag_old,
253  char *cnt,
254  unsigned char *par)
255 {
256  int val_diff;
257  bool flag_new;
258 
259  /* deal with SVM by limiting filter */
260  if (val_new > MAX_LIMIT_SVM) {
261  val_new = MAX_LIMIT_SVM;
262  }
263 
264  /* deal with SVM by low pass filter */
265  val_diff = val_new - val_old;
266 
267  if (val_diff) {
268  /* calculate low pass filter coefficient for SVM */
269  if (val_diff > 0) {
270  flag_new = true;
271  } else {
272  flag_new = false;
273  }
274 
275  if (flag_new == *flag_old) {
276  (*cnt)++;
277  if (fabs(val_diff) > THOLD_SVM_DIFF) {
278  *cnt += CNT_SVM_STEP;
279  }
280  if (*cnt > THOLD_SVM_CNT) {
281  *par += PAR_SVM_STEP;
282  *cnt = 0;
283  }
284  } else {
285  *cnt = 0;
286  *par = PAR_SVM_BASE;
287  }
288 
289  *flag_old = flag_new;
290  } else {
291  *cnt = 0;
292  *par = PAR_SVM_BASE;
293  }
294 
295  /* calculate output of low pass filter */
296  if (!val_diff) {
297  val_new = val_old;
298  } else {
299  val_new = val_old + val_diff * (*par) / 256;
300  }
301 
302  /* deal with SVM by limiting filter */
303  if (val_new < MIN_LIMIT_SVM) {
304  val_new = 0;
305  }
306 
307  return val_new;
308 }

311 static uint32_t process_acc(acc_values acc_temp)
312 {
313  int16_t x_new, y_new, z_new; /* latest value */
314  uint32_t svm_new; /* SVM : signal vector magnitude for difference */
315 
316  x_new = (int16_t)(100 * acc_temp.accl_x);
317  y_new = (int16_t)(100 * acc_temp.accl_y);
318  z_new = (int16_t)(100 * acc_temp.accl_z);
319 
320  /* deal with raw accelerate data by filter */
321  x_new = filter_acc(x_new, x_old, &flag_old_x, &cnt_x, &par_x);
322  y_new = filter_acc(y_new, y_old, &flag_old_y, &cnt_y, &par_y);
323  z_new = filter_acc(z_new, z_old, &flag_old_z, &cnt_z, &par_z);
324 
325  /* calculate SVM using data fusion */
326  svm_new = sqrt((x_new - x_old) * (x_new - x_old) +
327  (y_new - y_old) * (y_new - y_old) +
328  (z_new - z_old) * (z_new - z_old));
329 
330  x_old = x_new;
331  y_old = y_new;
332  z_old = z_new;
333 
334  /* deal with SVM by filter */
335  svm_new = filter_svm(svm_new, svm_old, &flag_old_v, &cnt_v, &par_v);
336  svm_old = svm_new;
337 
338  return svm_new;
339 }

342 static uint8_t func_detect_awake(uint32_t inten_temp)
343 {
344  bool flag_break_aw = false;
345  uint8_t cnt_sl_aw = 0;
346  uint8_t cnt_wk_aw = 0;
347  uint8_t event = NOEVENT; /* flag of event : NOEVENT or AWAKE */
348 
349  /* judge current status */
350  if (inten_temp > THOLD_INTEN_AW) {
351  state_aw[0] = 0; /* in wake state now */
352  } else {
353  state_aw[0] = 1; /* in sleep state now */
354  }
355 
356  /* the number of sleep state in front of the state queue */
357  for (uint8_t i = LEN_STA_QUEUE - 1; i > LEN_STA_WAKE - 1; --i) {
358  if (flag_break_aw) {
359  flag_break_aw = false;
360  break;
361  }
362 
363  if (state_aw[i]) {
364  cnt_sl_aw++;
365  }
366 
367  if (cnt_sl_aw > LEN_STA_SLEEP * PAR_STA_AW) {
368  cnt_sl_aw = 0;
369 
370  /* statistics the number of wake state in back of the state queue */
371  for (uint8_t i = 0; i < LEN_STA_WAKE; ++i) {
372  if (!state_aw[i]) {
373  cnt_wk_aw++;
374  }
375 
376  if (cnt_wk_aw > LEN_STA_WAKE * PAR_STA_AW) {
377  event = AWAKE; /* wake up event detected */
378  flag_break_aw = true; /* go exit from for loop */
379  } else {
380  event = NOEVENT;
381  }
382  }
383  } else {
384  event = NOEVENT;
385  }
386  }
387 
388  cnt_sl_aw = 0;
389  cnt_wk_aw = 0;
390 
391 #if PRINT_DEBUG_AWAKE
392  print_msg_awake(); /* print out message for debug */
393 #endif
394 
395  /* update motion intensity for LEN_STA_QUEUE * 5s */
396  for (uint8_t i = LEN_STA_QUEUE - 1; i > 0; --i) {
397  state_aw[i] = state_aw[i - 1];
398  }
399 
400  return event;
401 }

404 static uint8_t func_detect_state(uint32_t inten_temp)
405 {
406  uint8_t state; /* state : SLEEP or WAKE */
407 
408  inten_sl[0] = inten_temp; /* motion intensity in 1min */
409 
410  /* calculate the score of 2min ago(inten_sl[2]) by Webster sleep-wake determine method */
411  score_sl = P * (K4 * inten_sl[4] + K3 * inten_sl[3] +
412  K2 * inten_sl[2] + K1 * inten_sl[1] +
413  K0 * inten_sl[0]) / 100;
414 
415  /* determine the state of 2min ago by score */
416  if (score_sl > 1) {
417  state = WAKE;
418  } else {
419  state = SLEEP;
420  }
421 
422 #if PRINT_DEBUG_SLEEP
423  print_msg_sleep(state); /* print out message for debug */
424 #endif
425 
426  /* update motion intensity for 5 * 1min */
427  for (uint8_t i = 4; i > 0; --i) {
428  inten_sl[i] = inten_sl[i - 1];
429  }
430 
431  return state;
432 }
