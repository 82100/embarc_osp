124 static void task_process_hrate(void *par)
125 {
126  int32_t ir_value; /* IR value of heartrate sensor(max30102) */
127  int32_t data_rdy; /* flag of IR data ready */
128  uint32_t delta = 1; /* time interval between 2 heartbeats */
129 
130  for (;;) {
131  ir_value = 0;
132 
133  /* read IR data from heartrate sensor */
134  data_rdy = hrate_sensor_read(&ir_value);
135 
136  /* IR data ready and detect a heartbeat */
137  if (data_rdy == E_OK && ir_value != 0 && check_beat(ir_value) == true) {
138  flag_timer_stop = !flag_timer_stop;
139 
140  if (flag_timer_stop == true) {
141  delta = perf_end(); /* get cycles cost */
142 
143  beats_per_min = HRATE_BASE_VALUE / delta; /* get heartrate */
144 
145  if (beats_per_min < BEATS_PER_MIN_MAX && beats_per_min > BEATS_PER_MIN_MIN) {
146  /* sliding average filtering */
147  rates[rate_spot++] = (uint16_t)beats_per_min; /* store this reading in the array */
148  rate_spot %= HRATE_SIZE; /* wrap variable */
149 
150  /* take average of readings */
151  beat_aver = 0;
152  for (uint8_t x = 0 ; x < HRATE_SIZE ; x++)
153  beat_aver += rates[x];
154  beat_aver /= HRATE_SIZE;
155  }
156  } else {
157  perf_start(); /* get new counter number of t1 and flush t1_start */
158  }
159  }
160 
161  data_report_wn.hrate = beat_aver;
162 
163  data_report_wn.warn_hrate = false;
164  if (beat_aver < WARN_HR_MIN || beat_aver > WARN_HR_MAX) {
165  data_report_wn.warn_hrate = true;
166  }
167 
168  vTaskDelay(TIME_DELAY_HRATE_MS);
169  }
170 }

177 static bool check_beat(int32_t sample)
178 {
179  bool beat_detected = false;
180 
181  /* save current state */
182  val_pre = val_cur;
183 
184  /* process next data sample */
185  aver_estimated = aver_dc_estimator(&ir_avg_reg, sample);
186  val_cur = low_pass_fir_filter(sample - aver_estimated);
187 
188  /* detect positive zero crossing (rising edge) */
189  if ((val_pre < 0) & (val_cur >= 0)) {
190  ir_ac_max = val_max; /* adjust our AC max and min */
191  ir_ac_min = val_min;
192 
193  pos_edge = 1;
194  neg_edge = 0;
195  val_max = 0;
196 
197  if ((ir_ac_max - ir_ac_min) > AMPL_DIFF_MIN & (ir_ac_max - ir_ac_min) < AMPL_DIFF_MAX) {
198  /* heart beat detected */
199  beat_detected = true;
200  }
201  }
202 
203  /* detect negative zero crossing (falling edge) */
204  if ((val_pre > 0) & (val_cur <= 0)) {
205  pos_edge = 0;
206  neg_edge = 1;
207  val_min = 0;
208  }
209 
210  /* find maximum value in positive cycle */
211  if (pos_edge & (val_cur > val_pre)) {
212  val_max = val_cur;
213  }
214 
215  /* find minimum value in negative cycle */
216  if (neg_edge & (val_cur < val_pre)) {
217  val_min = val_cur;
218  }
219 
220  return (beat_detected);
221 }

231 static int16_t low_pass_fir_filter(int16_t din)
232 {
233  cbuf[offset] = din;
234 
235  int32_t z = mul16(fir_coeffs[11], cbuf[(offset - 11) & 0x1F]);
236 
237  for (uint8_t i = 0 ; i < 11 ; i++) {
238  z += mul16(fir_coeffs[i], cbuf[(offset - i) & 0x1F] +
239  cbuf[(offset - 22 + i) & 0x1F]);
240  }
241  offset++;
242  offset %= 32; /* wrap condition */
243 
244  return (z >> 15);
245 }
