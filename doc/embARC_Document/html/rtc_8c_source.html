172 uint8_t pcf8563_date_updata(uint8_t clock_day, uint8_t clock_hour, uint8_t clock_min)
173 {
174  uint8_t time[5];
175  iic0_point_rtc();
176  pcf8563_data_read(0x02, 0x05, time);
177 
178  /* If the num of second is zero,update the date again! */
179  if (rtc.sec == 0) {
180  pcf8563_data_read(0x02, 0x05, time);
181  }
182 
183  /* Process the data which read from the register */
184  /* The bits from 0 to 6 are effective */
185  rtc.sec = *(time + 0) & 0x7f; /* second */
186  /* The bits from 0 to 6 are effective */
187  rtc.min = *(time + 1) & 0x7f; /* minute */
188  /* The bits from 0 to 5 are effective */
189  rtc.hour = *(time + 2) & 0x3f; /* hour */
190  /* The bits from 0 to 3 are effective */
191  rtc.day = *(time + 4) & 0x07; /* week */
192 
193  pcf8563_bcd2de();
194  //EMBARC_PRINTF("Second:%d. Min:%d. hour:%d. day:%d.\r\n", rtc.sec, rtc.min, rtc.hour, rtc.day);
195 
196  if (rtc.hour == clock_hour) {
197  if (rtc.min == clock_min) {
198  return CLOCK_START;
199  }
200  }
201  return 0;
202 }
