114 static void pattern_change_task(void * par)
115 {
116  mode_info mode_info_now = INIT_MODE_INFO;
117  mode_info mode_info_old = INIT_MODE_INFO;
118  uint8_t mode_flag = ACTION_START;
119 
120  recognition_semaphore = xSemaphoreCreateBinary();
121  if (recognition_semaphore == NULL) {
122  EMBARC_PRINTF("Failed to create the recognition semaphore.\r\n");
123  vTaskSuspend(NULL);
124  }
125  mode_info_queue = xQueueCreate (1, sizeof(mode_info));
126  if (mode_info_queue == NULL) {
127  EMBARC_PRINTF("Failed to create the mode information queue.\r\n");
128  vTaskSuspend(NULL);
129  }
130  xQueueOverwrite(mode_info_queue, &mode_info_now);
131  EMBARC_PRINTF("Pattern change ready!\r\n");
132  vTaskSuspend(NULL);
133 
134  while(1) {
135  if (mode_info_old.mode == LIGHT_MODE_MUSIC) {
136  vTaskSuspend(light_mode_music_task_handle);
137  } else if (mode_info_old.mode >= MIN_MODE && mode_info_old.mode <= MAX_MODE) {
138  vTaskSuspend(light_mode_task_handle);
139  } else {
140  EMBARC_PRINTF("Mode error.\r\n");
141  }
142 
143  if (xSemaphoreTake(recognition_semaphore, (TickType_t)0) == pdTRUE) {
144  EMBARC_PRINTF("Recognitioning\r\n");
145  vTaskSuspend(NULL);
146  }
147 
148  light_ctr_reset();
149  xQueuePeek(mode_info_queue, &mode_info_now, (TickType_t)0);
150 
151  /* restart light mode task */
152  if (mode_info_now.mode == LIGHT_MODE_TIMING_FIT) {
153  time_clock = mode_info_now.time;
154  mode_info_now.mode = mode_info_old.mode;
155  } else if (mode_info_now.mode != mode_info_old.mode
156  || mode_info_now.time != mode_info_old.time
157  || mode_info_now.frequence != mode_info_old.frequence
158  || mode_info_now.mode == LIGHT_MODE_SHAKING
159  || mode_info_now.mode == LIGHT_MODE_TIMING_FIT) {
160  vTaskDelete(light_mode_task_handle);
161  if (xTaskCreate(light_mode_task, "light_mode_task", 128, (void *)1,
162  light_mode_task_prioroties, &light_mode_task_handle) != pdPASS) {
163  EMBARC_PRINTF("Restart light mode task error\r\n");
164  }
165  vTaskSuspend(light_mode_task_handle);
166  }
167 
168  if (mode_info_now.mode == LIGHT_MODE_MUSIC) {
169  vTaskSuspend(mpu_task_handle);
170  iic_point_mic();
171  vTaskResume(light_mode_music_task_handle);
172  EMBARC_PRINTF("Enable light mode music.\r\n");
173  } else if (mode_info_now.mode >= MIN_MODE && mode_info_now.mode <= MAX_MODE) {
174  iic_point_mpu();
175  vTaskResume(mpu_task_handle);
176  vTaskResume(light_mode_task_handle);
177  EMBARC_PRINTF("Enable light mode.\r\n");
178  } else {
179  EMBARC_PRINTF("The mode isn't changed.\r\n");
180  }
181  mode_info_old = mode_info_now;
182 
183  led_turn(0);
184  vTaskSuspend(NULL);
185  }
186 }

209 static void light_mode_music_task(void * par)
210 {
211  /* music data update and signal giver to light */
212  TickType_t xLastExecutionTime;
213  uint16_t fft_num = 0;
214  /* fft data segment */
215  compx fft_data[fft_N];
216  /* fft data input */
217  uint16_t fft_in[fft_N];
218  /* fft data output */
219  uint8_t fft_out[fft_N / 2];
220 
221  xLastExecutionTime = xTaskGetTickCount();
222  vTaskSuspend(NULL);
223 
224  while(1) {
225  fft_in[fft_num] = adc_get();
226  if (++fft_num >= fft_N) {
227  fft_num = 0;
228  fft_Data(fft_in, fft_data);
229  fft_Convert(fft_data); /* FFT resolve */
230  fft_powerMag(fft_data, fft_out); /* data compress */
231  light_mode_music(fft_out);
232  }
233  led_turn(2);
234  vTaskDelayUntil(&xLastExecutionTime, MIC_PERIOD_TICK);
235  }

267 static void mpu_task(void * par)
268 {
269  /* MPU read and imu update */
270  TickType_t xLastExecutionTime;
271  uint32_t recognition_cnt = 0;
272  mode_info mode_info_now = INIT_MODE_INFO;
273  uint8_t mode_flag = 0;
274  uint8_t recognition_flag = 0;
275  xLastExecutionTime = xTaskGetTickCount();
276 
277  while(1) {
278  imu_mpu6050_update();
279  recognition_cnt++;
280 
281  if (recognition_cnt == REC_PERIOD_TICK / MPU_PERIOD_TICK) {
282  recognition_cnt = 0;
283  mode_flag = imu_pattern_recongnition();
284  if (recognition_flag == 0 && mode_flag == ACTION_START) {
285  recognition_flag = 1;
286  light_mode_change_success();
287  xQueuePeek(mode_info_queue, &mode_info_now, (TickType_t)0);
288  light_mode_change_flag(mode_info_now.mode);
289  EMBARC_PRINTF("Start of recognition.\r\n");
290  xSemaphoreGive(recognition_semaphore);
291  vTaskResume(pattern_change_task_handle);
292  }
293  else if (recognition_flag == 1 && mode_flag == ACTION_RIGHT) {
294  mode_info_now.mode++;
295 
296  if (mode_info_now.mode > MAX_MODE){
297  mode_info_now.mode = MIN_MODE;
298  }
299 
300  light_mode_change_flag(mode_info_now.mode);
301  EMBARC_PRINTF("mode ++ %d.\r\n", mode_info_now.mode);
302  }
303  else if (recognition_flag == 1 && mode_flag == ACTION_LEFT) {
304  mode_info_now.mode--;
305 
306  if (mode_info_now.mode < MIN_MODE){
307  mode_info_now.mode = MAX_MODE;
308  }
309 
310  light_mode_change_flag(mode_info_now.mode);
311  EMBARC_PRINTF("mode -- %d.\r\n", mode_info_now.mode);
312  }
313  else if (recognition_flag == 1 && mode_flag == ACTION_OVER) {
314  recognition_flag = 0;
315 
316  if (mode_info_now.mode >= MIN_MODE && mode_info_now.mode <= MAX_MODE) {
317  xQueueOverwrite(mode_info_queue, &mode_info_now);
318  EMBARC_PRINTF("After pattern recognition mode is %d .\r\n", mode_info_now.mode);
319  } else {
320  EMBARC_PRINTF("Wrong pattern recognition mode is %d .\r\n", mode_info_now.mode);
321  }
322 
323  vTaskResume(pattern_change_task_handle);
324  }
325  }
326 
327  led_turn(1);
328  vTaskDelayUntil(&xLastExecutionTime, MPU_PERIOD_TICK);
329  /* This task should execute exactly every MPU_PERIOD_TICK */
330  }
331 }

412 static unsigned int perf_end(void)
413 {
414  unsigned int end = 0;
415 
416  if (timer_current(perf_id, (void *)(&end)) < 0) {
417  return 0;
418  }
419  if (start < end) {
420  return (end - start);
421  } else {
422  return (0xFFFFFFFF - start + end);
423  }
