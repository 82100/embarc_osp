113 extern uint32_t perf_end(void)
114 {
115  uint32_t t1_end = 0;
116 
117  if (timer_current(TIMER_1, (void *)(&t1_end)) < 0) {
118  return 0;
119  }
120 
121  if (t1_start < t1_end) {
122  return (t1_end - t1_start);

124  return (0xFFFFFFFF - t1_start + t1_end);
