122 uint8_t pmrf_read_long_ctrl_reg(uint16_t addr)
123 {
124  uint8_t msg[2];
125  uint8_t ret_val;
126  DEV_SPI_TRANSFER pmrf_xfer;
127  uint32_t cpu_status;
128 
129  msg[0] = (uint8_t)(((addr >> 3) & 0x7F) | 0x80);
130  msg[1] = (uint8_t)((addr << 5) & 0xE0);
131 
132  DEV_SPI_XFER_SET_TXBUF(&pmrf_xfer, msg, 0, 2);
133  DEV_SPI_XFER_SET_RXBUF(&pmrf_xfer, &ret_val, 2, 1);
134  DEV_SPI_XFER_SET_NEXT(&pmrf_xfer, NULL);
135 
136 #ifdef EMSK_PMRF_0_SPI_CPULOCK_ENABLE
137  cpu_status = cpu_lock_save();
138 #endif
139  pmrf_spi_ptr->spi_control(SPI_CMD_MST_SEL_DEV, CONV2VOID(cs_line));
140  pmrf_spi_ptr->spi_control(SPI_CMD_TRANSFER_POLLING, CONV2VOID(&pmrf_xfer));
141  pmrf_spi_ptr->spi_control(SPI_CMD_MST_DSEL_DEV, CONV2VOID(cs_line));
142 #ifdef EMSK_PMRF_0_SPI_CPULOCK_ENABLE
143  cpu_unlock_restore(cpu_status);
144 #endif
145 
146  return ret_val;
